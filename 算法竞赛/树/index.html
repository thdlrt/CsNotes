    push_up(p);
}
int query(int L,int R,int p,int pl,int pr){
    if(pl>=L && R >= pr)   return tree[p] %= mod;
    push_down(p,pl,pr);
    int res =0;
    int mid = (pl+pr) >> 1;
    if(L<=mid)   res += query(L,R,ls(p),pl,mid);
    if(R> mid)   res += query(L,R,rs(p),mid+1,pr);
    return res;
}

int son[N],id[N],fa[N],deep[N],siz[N],top[N];
void dfs1(int x, int father){  
    deep[x]=deep[father]+1;   //ж·ұеәҰпјҡжҜ”зҲ¶з»“зӮ№ж·ұеәҰеӨҡ1
fa[x]=father;             //ж Үи®°xзҡ„зҲ¶дәІ
    siz[x]=1;                 //ж Үи®°жҜҸдёӘз»“зӮ№зҡ„еӯҗж ‘еӨ§е°ҸпјҲеҢ…жӢ¬иҮӘе·ұпјү
    for(int i=head[x];~i;i=edge[i].next){
        int y=edge[i].to;
        if(y!=father){   //йӮ»еұ…пјҡйҷӨдәҶзҲ¶дәІпјҢйғҪжҳҜеӯ©еӯҗ
           fa[y]=x;
           dfs1(y,x);
           siz[x] += siz[y];     //еӣһжәҜеҗҺпјҢжҠҠxзҡ„е„ҝеӯҗж•°еҠ еҲ°xиә«дёҠ
           if(!son[x] || siz[son[x]]<siz[y])   //ж Үи®°жҜҸдёӘйқһеҸ¶еӯҗз»“зӮ№зҡ„йҮҚе„ҝеӯҗ
               son[x]=y;                       //xзҡ„йҮҚе„ҝеӯҗжҳҜy
        }
    }
}
int num = 0;
void dfs2(int x,int topx){    //xеҪ“еүҚз»“зӮ№пјҢtopxеҪ“еүҚй“ҫзҡ„жңҖйЎ¶з«Ҝзҡ„з»“зӮ№
    id[x] = ++num;            //еҜ№жҜҸдёӘз»“зӮ№ж–°зј–еҸ·

    top[x]=topx;              //и®°еҪ•xзҡ„й“ҫеӨҙ
    if(!son[x])   return;     //xжҳҜеҸ¶еӯҗпјҢжІЎжңүе„ҝеӯҗпјҢиҝ”еӣһ
    dfs2(son[x],topx);                     //е…ҲdfsйҮҚе„ҝеӯҗ
    for(int i=head[x];~i;i=edge[i].next){  //еҶҚdfsиҪ»е„ҝеӯҗ
        int y=edge[i].to;
        if(y!=fa[x] && y!=son[x]) dfs2(y,y);//жҜҸдёӘиҪ»е„ҝеӯҗйғҪжңүдёҖжқЎд»Һе®ғиҮӘе·ұејҖе§Ӣзҡ„й“ҫ
    }
}
void update_range(int x,int y,int z){        //е’ҢжұӮLCA(x, y)зҡ„иҝҮзЁӢе·®дёҚеӨҡ
    while(top[x]!=top[y]){
        if(deep[top[x]]<deep[top[y]])   swap(x,y);
        update(id[top[x]],id[x],1,1,n,z);    //дҝ®ж”№дёҖжқЎйҮҚй“ҫзҡ„еҶ…йғЁ
        x = fa[top[x]];
    }
    if(deep[x]>deep[y])    swap(x,y);
    update(id[x],id[y],1,1,n,z);             //дҝ®ж”№дёҖжқЎйҮҚй“ҫзҡ„еҶ…йғЁ
}
int query_range(int x,int y){                //е’ҢжұӮLCA(x,y)зҡ„иҝҮзЁӢе·®дёҚеӨҡ
    int ans=0;
    while(top[x]!=top[y]){                   //жҢҒз»ӯеҫҖдёҠи·іпјҢзӣҙеҲ°иӢҘxе’ҢyеұһдәҺеҗҢдёҖжқЎйҮҚй“ҫ
        if(deep[top[x]]<deep[top[y]]) swap(x,y); //и®©xжҳҜй“ҫеӨҙжӣҙж·ұзҡ„йҮҚй“ҫ
        ans += query(id[top[x]],id[x],1,1,n);    //еҠ дёҠxеҲ°xзҡ„й“ҫеӨҙиҝҷдёҖж®өеҢәй—ҙ
        ans %= mod;
        x = fa[top[x]];                          //xз©ҝиҝҮиҪ»иҫ№пјҢи·іеҲ°дёҠдёҖжқЎйҮҚй“ҫ
    }
if(deep[x]>deep[y])  swap(x,y);

    return ans % mod;
}
